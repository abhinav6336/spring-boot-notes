# The th:if and th:unless attributes are used for rendering the contents on Thymeleaf template according to the condition.

Syntax

```html
<div th:if="condition">
	<p>True Statement</p>
</div>
<div th:unless="condition">
	<p> False Statement</p>
</div>
```

example

```java
@GetMapping("/home")
	public String home(Model model)
	{
		model.addAttribute("username","Dhananjay");
		model.addAttribute("usertype","admin");
		return "home";
	}
```

inside home.html

```html
<div th:if="${usertype}=='admin'">
	<p style="color: green;">Welcome <span th:text="${username}"></span></p>
</div>
<div th:unless="${usertype}=='admin'">
	<p style="color: red;">Welcome Guest... </p>
</div>
```

# The switch-case condition in Thymeleaf template

```html
<div th:switch="expression">
	<p th:case="'value-1'"> statement-1</p>
	<p th:case="'value-1'"> statement-1</p>
	____________________
	____________________
	<p th:case="*">default statement</p>
</div>
```

example

```html
<div th:switch="${usertype}">
	<p th:case="'admin'" style="color: green;"> Wewlcome <span th:text="${username}"></span></p>
	<p th:case="'staff'" style="color: blue;"> Staff Message</p>
	<p th:case="*" style="color: red;">Unknown guest</p>
</div>
```

```java

```
	int rollno;
	String name;
	String course;
	String branch;
	public Student() {}
	public Student(int rollno, String name, String course, String branch) {
		super();
		this.rollno = rollno;
		this.name = name;
		this.course = course;
		this.branch = branch;
	}
	public int getRollno() {
		return rollno;
	}
	public String getName() {
		return name;
	}
	public String getCourse() {
		return course;
	}
	public String getBranch() {
		return branch;
	}
	public void setRollno(int rollno) {
		this.rollno = rollno;
	}
	public void setName(String name) {
		this.name = name;
	}
	public void setCourse(String course) {
		this.course = course;
	}
	public void setBranch(String branch) {
		this.branch = branch;
	}
