Return Class components through API

1. Here we can treat a class as component and defined it inside Component Package
2. Create a package named as component
3. for example we want to process data of a students, so create a class named as student inside component package.
package com.training.Component;

import org.springframework.stereotype.Component;

@Component
public class student {
	int rollno,fees;
	String name,course;
	public student()
	{}
	public student(int rollno, int fees, String name, String course) {
		super();
		this.rollno = rollno;
		this.fees = fees;
		this.name = name;
		this.course = course;
	}
	public int getRollno() {
		return rollno;
	}
	public int getFees() {
		return fees;
	}
	public String getName() {
		return name;
	}
	public String getCourse() {
		return course;
	}
	public void setRollno(int rollno) {
		this.rollno = rollno;
	}
	public void setFees(int fees) {
		this.fees = fees;
	}
	public void setName(String name) {
		this.name = name;
	}
	public void setCourse(String course) {
		this.course = course;
	}
	
}

4. now create an api-
@GetMapping("/stuinfo")
	public student stuinfo()
	{
		return new student(101, 120000, "Anuj Kapoor", "B.Tech.");
	}
5. finally call it
http://localhost:8888/stuinfo

create an api named whomi() which contains name and address and print them on browser.

@GetMapping("/whomi/{name}/{address}")
	public String whomi(@PathVariable String name,@PathVariable String address)
	{
		String msg="Name="+name+"<br>";
		msg+="Address="+address;
		return msg;
	}
